35def betastar_from_phase(meas_input: DotDict, phase_d: phase.PhaseDict) -> pd.DataFrame: 

36 """ 

37 Calculate beta* and l* from the phase advance of the IP-BPMs. 

38 

39 Arguments: 

40 meas_input: `Measurement_input` object. 

41 phase_d: PhaseDict output of the phasecalculation. 

42 

43 Returns: 

44 A DataFrame with the beta* and l* as well as the phases used for the calculation 

45 as columns and the IP names as index. 

46 """ 

47 accel = meas_input.accelerator 

48 model = accel.model 

49 try: 

50 ips = list(accel.get_ips()) 

51 except AttributeError: 

52 LOGGER.debug(f"Accelerator {type(accel).__name__} has not get_ips method.") 

53 return None 

54 rows = [] 

55 for ip_name, bpml, bpmr in ips: 

56 try: 

57 phaseadv, ephaseadv, mdlphadv = _get_meas_phase(bpml, bpmr, phase_d) 

58 except KeyError: 

59 LOGGER.debug(f"{bpml} {bpmr} not in phases dataframe.") 

60 continue # Measurement on one of the BPMs not present. 

61 lstar = _get_lstar(bpml, bpmr, model) 

62 betastar, ebestar = phase_to_betastar(lstar, PI2 * phaseadv, PI2 * ephaseadv) 

63 rows.append([ip_name, betastar, ebestar, phaseadv, ephaseadv, mdlphadv, lstar]) 

64 return pd.DataFrame(columns=COLUMNS, data=rows)

82def phase_to_betastar(lstar: float, phase: float, errphase: float) -> tuple[float, float]: 

83 """ 

84 Return the betastar and its error given the phase advance across the IP. 

85 

86 This function computes the betastar using the phase advance between the BPMs around the IP 

87 and their distance (lstar). The phase and error in the phase must be given in radians. 

88 

89 Args: 

90 lstar (float): The distance between the BPMs and the IR. 

91 phase (float): The phase advance between the BPMs at each side of the IP. Must be given in radians. 

92 errphase (float): The error in phase. Must be given in radians. 

93 

94 Returns: 

95 tuple[float, float]: betastar and its error. 

96 """ 

97 return ( 

98 _phase_to_betastar_value(lstar, phase), 

99 _phase_to_betastar_error(lstar, phase, errphase), 

100 ) 

101 

102 

103def _phase_to_betastar_value(lstar, phase): 

104 tan_phase = np.tan(phase) 

105 return (lstar * (1 - np.sqrt(tan_phase ** 2 + 1))) / tan_phase 

106 

107 

108def _phase_to_betastar_error(lstar: float, phase: float, errphase: float): 

109 return abs((errphase * lstar * (abs(np.cos(phase)) - 1)) / (np.sin(phase) ** 2))

112def _get_meas_phase(bpm_left, bpm_right, phases_df): 

113 return ( 

114 phases_df[MEASUREMENT].loc[bpm_left, bpm_right], 

115 phases_df[f"{ERR}{MEASUREMENT}"].loc[bpm_left, bpm_right], 

116 phases_df[MODEL].loc[bpm_left, bpm_right], 

117 )

120def _get_lstar(bpm_left, bpm_right, model): 

121 return abs(model.loc[bpm_left, S] - model.loc[bpm_right, S]) / 2.
