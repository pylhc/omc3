59def run_per_bunch( 

60 tbt_data: TbtData, harpy_input: DotDict, output_filename: str 

61) -> dict[str, tfs.TfsDataFrame]: 

62 """ 

63 Cleans data, analyses frequencies and searches for resonances. 

64 

65 Args: 

66 tbt_data: single bunch `TbtData`. 

67 harpy_input: Analysis settings taken from the commandline. 

68 output_filename: Name of the output files (placed in `harpy_input.outputdir`). 

69 

70 Returns: 

71 Dictionary with a `TfsDataFrame` per plane. 

72 """ 

73 model = None 

74 if harpy_input.model is not None: 

75 model = tfs.read(harpy_input.model, index=COL_NAME).loc[:, COL_S] 

76 

77 bpm_datas, usvs, lins, bad_bpms = {}, {}, {}, {} 

78 output_file_path = harpy_input.outputdir / output_filename 

79 

80 for plane in PLANES: 

81 bpm_data = _get_cut_tbt_matrix(tbt_data, harpy_input.turns, plane) 

82 bpm_data = _scale_to_meters(bpm_data, harpy_input.unit) 

83 bpm_data, usvs[plane], bad_bpms[plane], bpm_res = clean.clean(harpy_input, bpm_data, model) 

84 lins[plane], bpm_datas[plane] = _closed_orbit_analysis(bpm_data, model, bpm_res) 

85 

86 tune_estimates = ( 

87 harpy_input.tunes 

88 if harpy_input.autotunes is None 

89 else frequency.estimate_tunes( 

90 harpy_input, 

91 usvs 

92 if harpy_input.clean 

93 else { 

94 "X": clean.svd_decomposition(bpm_datas["X"], harpy_input.sing_val), 

95 "Y": clean.svd_decomposition(bpm_datas["Y"], harpy_input.sing_val), 

96 }, 

97 ) 

98 ) 

99 

100 spectra = {} 

101 for plane in PLANES: 

102 with timeit(lambda spanned: LOGGER.debug(f"Time for harmonic_analysis: {spanned}")): 

103 harpy_results, spectra[plane], bad_bpms_summaries = frequency.harpy_per_plane( 

104 harpy_input, bpm_datas[plane], usvs[plane], tune_estimates, plane 

105 ) 

106 if "bpm_summary" in harpy_input.to_write: 

107 bad_bpms[plane].extend(bad_bpms_summaries) 

108 _write_bad_bpms(output_file_path, plane, bad_bpms[plane]) 

109 

110 if "spectra" in harpy_input.to_write or "full_spectra" in harpy_input.to_write: 

111 _write_spectrum(output_file_path, plane, spectra[plane]) 

112 

113 lins[plane] = lins[plane].loc[harpy_results.index].join(harpy_results) 

114 

115 if harpy_input.is_free_kick: 

116 # Free kick assumes exponentially decaying oscillations. 

117 lins[plane] = kicker.phase_correction(bpm_datas[plane], lins[plane], plane) 

118 

119 measured_tunes = [ 

120 lins["X"][f"{COL_TUNE}X"].mean(), 

121 lins["Y"][f"{COL_TUNE}Y"].mean(), 

122 lins["X"][f"{COL_TUNE}Z"].mean() if tune_estimates[2] > 0 else 0, 

123 ] 

124 

125 for plane in PLANES: 

126 lins[plane] = lins[plane].join( 

127 frequency.find_resonances( 

128 tunes=measured_tunes, 

129 nturns=bpm_datas[plane].shape[1], 

130 plane=plane, 

131 spectra=spectra[plane], 

132 order_resonances=harpy_input.resonances, 

133 ) 

134 ) 

135 lins[plane] = _add_calculated_phase_errors(lins[plane]) 

136 lins[plane] = _sync_phase(lins[plane], plane) 

137 lins[plane] = _rescale_amps_to_main_line_and_compute_noise(lins[plane], plane) 

138 lins[plane] = lins[plane].sort_values(COL_S, axis=0, ascending=True) 

139 lins[plane] = tfs.TfsDataFrame( 

140 lins[plane], headers=_compute_headers(lins[plane], tbt_data.meta.get("date")) 

141 ) 

142 if "lin" in harpy_input.to_write: 

143 _write_lin_tfs(output_file_path, plane, lins[plane]) 

144 return lins

147def _get_cut_tbt_matrix(tbt_data: TbtData, turn_indices: list[int], plane: str) -> pd.DataFrame: 

148 start = max(0, min(turn_indices)) 

149 end = min(max(turn_indices), tbt_data.matrices[0][plane].shape[1]) 

150 return tbt_data.matrices[0][plane].iloc[:, start:end].T.reset_index(drop=True).T

153def _scale_to_meters(bpm_data: pd.DataFrame, unit: str) -> pd.DataFrame: 

154 scales_to_meters = {"um": 1e-6, "mm": 0.001, "cm": 0.01, "m": 1} 

155 bpm_data.iloc[:, :] = bpm_data.iloc[:, :].to_numpy() * scales_to_meters[unit] 

156 return bpm_data

159def _closed_orbit_analysis( 

160 bpm_data: pd.DataFrame, model: None | pd.DataFrame, bpm_res 

161) -> tuple[pd.DataFrame, pd.DataFrame]: 

162 # Initialize the linear frame DataFrame with BPM names and positions 

163 lin_frame = pd.DataFrame( 

164 index=bpm_data.index.to_numpy(), 

165 data={ 

166 COL_NAME: bpm_data.index.to_numpy(), 

167 COL_S: np.arange(bpm_data.index.size) if model is None else model.loc[bpm_data.index], 

168 }, 

169 ) 

170 lin_frame[COL_BPM_RES] = 0.0 if bpm_res is None else bpm_res.loc[lin_frame.index] 

171 with timeit(lambda spanned: LOGGER.debug(f"Time for orbit_analysis: {spanned}")): 

172 lin_frame = _get_orbit_data(lin_frame, bpm_data) 

173 

174 # Subtract the mean from BPM data 

175 mean_subtracted_bpm_data = bpm_data.subtract(bpm_data.mean(axis=1), axis=0) 

176 

177 return lin_frame, mean_subtracted_bpm_data

180def _get_orbit_data(lin_frame: pd.DataFrame, bpm_data: pd.DataFrame) -> pd.DataFrame: 

181 lin_frame[COL_PK2PK] = np.max(bpm_data, axis=1) - np.min(bpm_data, axis=1) 

182 lin_frame[COL_CO] = np.mean(bpm_data, axis=1) 

183 lin_frame[COL_CORMS] = np.std(bpm_data, axis=1) / np.sqrt(bpm_data.shape[1]) 

184 # TODO: Magic number 10?: Maybe accelerator dependent ... LHC 6-7? 

185 lin_frame[COL_NOISE] = lin_frame.loc[:, COL_BPM_RES] / np.sqrt(bpm_data.shape[1]) / 10.0 

186 return lin_frame

189def _add_calculated_phase_errors(lin_frame: pd.DataFrame) -> pd.DataFrame: 

190 """ 

191 Add calculated phase errors to the linear frame for all MU and PHASE columns. 

192 

193 For each column starting with COL_MU or COL_PHASE, computes the corresponding 

194 error column using the spectral phase error formula. 

195 """ 

196 if lin_frame[COL_NOISE].max() == 0.0: 

197 return lin_frame # Skip if no noise data available 

198 

199 noise_values = lin_frame[COL_NOISE].to_numpy() 

200 

201 # Get all columns that represent phases (MU or PHASE) 

202 phase_columns = [col for col in lin_frame.columns if col.startswith((COL_MU, COL_PHASE))] 

203 

204 for phase_col in phase_columns: 

205 # Corresponding amplitude column (e.g., MUX -> AMPX) 

206 amplitude_col = phase_col.replace(COL_MU, COL_AMP).replace(COL_PHASE, COL_AMP) 

207 

208 # Calculate and assign the error 

209 lin_frame[f"{COL_ERR}{phase_col}"] = _get_spectral_phase_error( 

210 amplitude=lin_frame[amplitude_col], 

211 noise=noise_values, 

212 ) 

213 

214 return lin_frame

217def _get_spectral_phase_error(amplitude: np.ndarray, noise: np.ndarray) -> np.ndarray: 

218 """ 

219 When the error is too big (> 2*pi*0.25 more or less) the noise is not Gaussian anymore. 

220 In such a case the distribution is almost uniform, so we set the error to be 0.3, which is 

221 the standard deviation of uniformly distributed phases. 

222 This approximation does not bias the error by more than 20%, and that is only for large errors. 

223 """ 

224 error = noise / (np.where(amplitude > 0.0, amplitude, 1e-15) * 2 * np.pi) 

225 return np.where(error > 0.25, 0.3, error)

228def _sync_phase(lin_frame: pd.DataFrame, plane: str) -> pd.DataFrame: 

229 """ 

230 Produces ``MUXSYNC`` and ``MUYSYNC`` columns that are ``MUX/MUY`` but shifted such that for 

231 BPM at index 0 is always 0. It allows to compare phases of consecutive measurements and if 

232 some measurements stick out remove them from the data set. Original author is **skowron**. 

233 """ 

234 phase = lin_frame[f"{COL_MU}{plane}"].to_numpy() 

235 phase = phase - phase[0] 

236 lin_frame[f"{COL_MU}{plane}SYNC"] = np.where(np.abs(phase) > 0.5, phase - np.sign(phase), phase) 

237 return lin_frame

240def _compute_headers(panda: pd.DataFrame, date: None | pd.Timestamp = None) -> dict[str, float]: 

241 headers = {} 

242 for plane in ALL_PLANES: 

243 for prefix in ("", "NAT"): 

244 try: 

245 bpm_tunes = panda[f"{prefix}{COL_TUNE}{plane}"] 

246 except KeyError: 

247 pass 

248 else: 

249 headers[f"{prefix}Q{PLANE_TO_NUM[plane]}"] = np.mean(bpm_tunes) 

250 headers[f"{prefix}Q{PLANE_TO_NUM[plane]}RMS"] = np.std( 

251 bpm_tunes 

252 ) # TODO: not really the RMS? 

253 if date: 

254 headers[COL_TIME] = date.strftime(formats.TIME) 

255 headers[MAINLINE_UNIT] = "m" 

256 return headers

286def _rescale_amps_to_main_line_and_compute_noise(df: pd.DataFrame, plane: str) -> pd.DataFrame: 

287 """ 

288 Rescale secondary amplitudes to main line amplitude and compute noise-related errors. 

289 """ 

290 cols = [col for col in df.columns if col.startswith(COL_AMP)] 

291 cols.remove(f"{COL_AMP}{plane}") 

292 df[cols] = df[cols].div(df[f"{COL_AMP}{plane}"], axis="index") 

293 amps = df[f"{COL_AMP}{plane}"].to_numpy() 

294 

295 if df[COL_NOISE].max() == 0.0: 

296 return df # Do not calculate errors when no noise was calculated 

297 noise_scaled = df[COL_NOISE] / amps 

298 df[COL_NOISE_SCALED] = noise_scaled 

299 df[f"{COL_ERR}{COL_AMP}{plane}"] = df[COL_NOISE] 

300 if f"{COL_NATTUNE}{plane}" in df.columns: 

301 df[f"{COL_ERR}{COL_NATAMP}{plane}"] = df[COL_NOISE] 

302 

303 # Create dedicated dataframe with error columns to assign later (cleaner 

304 # and faster than assigning individual columns) 

305 df_amp = pd.DataFrame( 

306 data={ 

307 f"{COL_ERR}{col}": noise_scaled * np.sqrt(1 + np.square(df.loc[:, col])) for col in cols 

308 }, 

309 index=df.index, 

310 dtype=pd.Float64Dtype(), 

311 ) 

312 df[df_amp.columns] = df_amp 

313 return df
