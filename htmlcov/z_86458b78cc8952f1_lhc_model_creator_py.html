92 """Use the fetcher to list choices if requested.""" 

93 accel: Lhc = self.accel 

94 

95 # Set the fetcher paths --- 

96 if opt.fetch == Fetcher.PATH: 

97 if opt.path is None: 

98 raise AcceleratorDefinitionError("Path fetcher chosen, but no path provided.") 

99 acc_model_path = Path(opt.path) 

100 

101 elif opt.fetch == Fetcher.AFS: 

102 # list 'year' choices --- 

103 acc_model_path = check_folder_choices( 

104 AFS_ACCELERATOR_MODEL_REPOSITORY / accel.NAME, 

105 msg="No optics tag (flag --year) given", 

106 selection=accel.year,

108 predicate=Path.is_dir, 

109 ) # raises AcceleratorDefintionError if not valid choice 

110 else: 

111 LOGGER.warning( 

112 f"{accel.NAME} model creation is expected to run via a fetcher these days. " 

113 "If you are creating an older model, this might all be correct " 

114 "and you can ignore this warning. Otherwise you will soon run into " 

115 "a MAD-X error. In this case, please provide a fetcher for the model via --fetch flag. " 

116 ) 

117 return

119 # list optics choices --- 

120 if opt.list_choices: 

121 # assumes we only want to list optics. Invoked even if modifiers are given! 

122 check_folder_choices( 

123 acc_model_path / OPTICS_SUBDIR, 

124 msg="No modifier given", 

125 selection=None, # TODO: could check if user made valid choice 

126 list_choices=opt.list_choices, 

127 predicate=get_check_suffix_func(".madx"), 

128 ) # raises AcceleratorDefinitionError

182 def get_madx_script(self) -> str: # nominal 

183 """Get madx script to create a LHC model.""" 

184 accel: Lhc = self.accel 

185 

186 madx_script = self.get_base_madx_script() 

187 twiss_dat_path = self.resolve_path_for_madx(accel.model_dir / TWISS_DAT) 

188 twiss_elements_path = self.resolve_path_for_madx(accel.model_dir / TWISS_ELEMENTS_DAT) 

189 madx_script += ( 

190 f"exec, do_twiss_monitors(LHCB{accel.beam}, '{twiss_dat_path}', {accel.dpp});\n" 

191 f"exec, do_twiss_elements(LHCB{accel.beam}, '{twiss_elements_path}', {accel.dpp});\n" 

192 ) 

193 if accel.excitation != AccExcitationMode.FREE or accel.drv_tunes is not None: 

194 # allow user to modify script and enable excitation, if driven tunes are given 

195 use_acd = accel.excitation == AccExcitationMode.ACD 

196 use_adt = accel.excitation == AccExcitationMode.ADT 

197 twiss_ac_path = self.resolve_path_for_madx(accel.model_dir / TWISS_AC_DAT) 

198 twiss_adt_path = self.resolve_path_for_madx(accel.model_dir / TWISS_ADT_DAT) 

199 madx_script += ( 

200 f"use_acd={use_acd:d};\n" 

201 f"use_adt={use_adt:d};\n" 

202 f"if(use_acd == 1){{\n" 

203 f"exec, twiss_ac_dipole({accel.nat_tunes[0]}, {accel.nat_tunes[1]}, {accel.drv_tunes[0]}, {accel.drv_tunes[1]}, {accel.beam}, '{twiss_ac_path}', {accel.dpp});\n" 

204 f"}}else if(use_adt == 1){{\n" 

205 f"exec, twiss_adt({accel.nat_tunes[0]}, {accel.nat_tunes[1]}, {accel.drv_tunes[0]}, {accel.drv_tunes[1]}, {accel.beam}, '{twiss_adt_path}', {accel.dpp});\n" 

206 f"}}\n" 

207 ) 

208 return madx_script

210 def get_base_madx_script(self) -> str: 

211 """Returns the base LHC madx script.""" 

212 accel: Lhc = self.accel 

213 

214 madx_script = self._get_sequence_initialize_script() 

215 if self._uses_ats_knobs(): 

216 madx_script += "\n! ----- Matching Knobs -----\n" 

217 LOGGER.debug( 

218 "According to the optics year or the --ats flag being provided, ATS macros and knobs will be used" 

219 ) 

220 madx_script += f"exec, match_tunes_ats({accel.nat_tunes[0]}, {accel.nat_tunes[1]}, {accel.beam});\n" 

221 madx_script += f"exec, coupling_knob_ats({accel.beam});\n" 

222 else: 

223 madx_script += ( 

224 f"exec, match_tunes({accel.nat_tunes[0]}, {accel.nat_tunes[1]}, {accel.beam});\n" 

225 ) 

226 madx_script += f"exec, coupling_knob({accel.beam});\n" 

227 

228 return madx_script

230 def _get_sequence_initialize_script(self) -> str: 

231 """Returns the LHC sequence initialization script. 

232 

233 This is split up here from the matching (in the base-script), 

234 to accompany the needs of the Best Knowledge Model Creator, 

235 see below. 

236 """ 

237 accel: Lhc = self.accel 

238 madx_script = ( 

239 f"{self._get_madx_script_info_comments()}\n\n" 

240 f"call, file = '{self.resolve_path_for_madx(accel.model_dir / MACROS_DIR / GENERAL_MACROS)}';\n" 

241 f"call, file = '{self.resolve_path_for_madx(accel.model_dir / MACROS_DIR / LHC_MACROS)}';\n" 

242 ) 

243 madx_script += f"{MADX_ENERGY_VAR} = {accel.energy};\n" 

244 madx_script += "exec, define_nominal_beams();\n\n" 

245 if self._uses_run3_macros(): 

246 LOGGER.debug("According to the optics year, Run 3 versions of the macros will be used") 

247 madx_script += f"call, file = '{self.resolve_path_for_madx(accel.model_dir / MACROS_DIR / LHC_MACROS_RUN3)}';\n" 

248 

249 madx_script += "\n! ----- Calling Sequence -----\n" 

250 madx_script += ( 

251 "option, -echo; ! suppress output from base sequence loading to keep the log small\n" 

252 ) 

253 madx_script += self._get_madx_main_sequence_loading() 

254 madx_script += "\noption, echo; ! re-enable output to see the optics settings\n" 

255 

256 madx_script += "\n! ---- Call optics and other modifiers ----\n" 

257 

258 if accel.modifiers is not None: 

259 # if the modifier can be found in the model dir, use relative path 

260 madx_script += "".join( 

261 f"call, file = '{self.resolve_path_for_madx(modifier)}'; {MODIFIER_TAG}\n" 

262 for modifier in accel.modifiers 

263 ) 

264 

265 if accel.year in ["2012", "2015", "2016", "2017", "2018", "2021", "hllhc1.3"]: 

266 # backwards compatibility with pre acc-models optics 

267 # WARNING: This might override values extracted via the knob-extractor. 

268 madx_script += ( 

269 f"\n! ----- Defining Configuration Specifics -----\n" 

270 f"xing_angles = {'1' if accel.xing else '0'};\n" 

271 f"if(xing_angles==1){{\n" 

272 f" exec, set_crossing_scheme_ON();\n" 

273 f"}}else{{\n" 

274 f" exec, set_default_crossing_scheme();\n" 

275 f"}}\n" 

276 ) 

277 

278 if accel.acc_model_path is not None: 

279 remove_symmetry_knob_abs = ( 

280 Path(accel.acc_model_path) / LHC_REMOVE_TRIPLET_SYMMETRY_RELPATH 

281 ) 

282 if remove_symmetry_knob_abs.exists(): # alternatively check if year != 2018/2021 

283 remove_symmetry_knob_path = self.resolve_path_for_madx(remove_symmetry_knob_abs) 

284 madx_script += ( 

285 "\n! ----- Remove IR symmetry definitions -----\n" 

286 f'\ncall, file="{remove_symmetry_knob_path}"; ' 

287 "! removes 'ktqx.r1 := -ktqx.l1'-type issues\n" 

288 ) 

289 

290 madx_script += ( 

291 "\n! ----- Finalize Sequence -----\n" 

292 "exec, cycle_sequences();\n" 

293 f"use, sequence = LHCB{accel.beam};\n" 

294 ) 

295 return madx_script

297 def get_update_deltap_script(self, deltap: float | str) -> str: 

298 """Update the dpp in the LHC. 

299 

300 Args: 

301 deltap (float | str): The dpp to update the LHC to. 

302 """ 

303 accel: Lhc = self.accel 

304 if not isinstance(deltap, str): 

305 deltap = f"{deltap:.15e}" 

306 

307 madx_script = ( 

308 f"twiss, deltap={deltap};\n" 

309 "correct, mode=svd;\n\n" 

310 "! The same as match_tunes, but include deltap in the matching\n" 

311 f"exec, find_complete_tunes({accel.nat_tunes[0]}, {accel.nat_tunes[1]}, {accel.beam});\n" 

312 f"match, deltap={deltap};\n" 

313 ) # Works better when split up 

314 madx_script += "\n".join([f"vary, name={knob};" for knob in self.get_tune_knobs()]) + "\n" 

315 madx_script += ( 

316 "constraint, range=#E, mux=total_qx, muy=total_qy;\n" 

317 "lmdif, tolerance=1e-10;\n" 

318 "endmatch;\n" 

319 ) 

320 return madx_script

379 def _uses_ats_knobs(self) -> bool: 

380 """ 

381 Returns wether the ATS knobs and macros should be used, based on the accel-instance's year. 

382 If the **--ats** flag was explicitely provided then the returned value will be `True`. 

383 """ 

384 try: 

385 if self.accel.ats: 

386 return True 

387 return 2018 <= int(self.accel.year) <= 2021 # self.year is always a string 

388 except ValueError: # if a "hllhc1.x" version is given 

389 return False 

390 

391 def _uses_run3_macros(self) -> bool: 

392 """Returns whether the Run 3 macros should be called, based on the accel-instance's year.""" 

393 try: 

394 return int(self.accel.year) >= 2022 # self.year is always a string 

395 except ValueError: # if a "hllhc1.x" year is given 

396 return False

511 f"exec, do_twiss_elements(LHCB{accel.beam}, '{twiss_bk_elements_path}', {accel.dpp});\n" 

512 )

531class LhcCorrectionModelCreator(CorrectionModelCreator, LhcModelCreator): # ----------------------- 

532 """ 

533 Creates an updated model from multiple changeparameters inputs 

534 (used in iterative correction). 

535 """ 

536 

537 def __init__( 

538 self, 

539 accel: Lhc, 

540 twiss_out: Path | str, 

541 corr_files: Sequence[Path | str], 

542 update_dpp: bool = False, 

543 ): 

544 """Model creator for the corrected/matched model of the LHC. 

545 

546 Inheritance (i.e. __init__ calls) from here should be as follows: 

547 This super() calls CorrectionModelCreator. 

548 The super() there calls LhcModelCreator which calls ModelCreator, setting the base-attributes. 

549 Then Attributes in CorrectionModelCreator are set. 

550 

551 Args: 

552 accel (Lhc): Accelerator Class Instance 

553 twiss_out (Union[Path, str]): Path to the twiss(-elements) file to write 

554 change_params (Sequence[Path]): Sequence of correction/matching files 

555 """ 

556 LOGGER.debug("Initializing LHC Correction Model Creator") 

557 super().__init__(accel, twiss_out, corr_files, update_dpp) 

558 

559 def get_madx_script(self) -> str: 

560 """Get the madx script for the correction model creator, which updates the model after correcion.""" 

561 accel: Lhc = self.accel 

562 

563 # do not super().get_madx_script as we don't need the uncorrected output. 

564 madx_script = self.get_base_madx_script() 

565 

566 # First set the dpp to the value in the accelerator model 

567 madx_script += f"{ORBIT_DPP} = {accel.dpp};\n" 

568 

569 for corr_file in self.corr_files: # Load the corrections, can also update ORBIT_DPP 

570 # Should be the correct path already from the init 

571 madx_script += f"call, file = '{corr_file}';\n" 

572 

573 # If we are doing orbit correction, we need to ensure that a correct and a match is done 

574 if self.update_dpp: 

575 madx_script += self.get_update_deltap_script(deltap=ORBIT_DPP) 

576 

577 twiss_out_path = self.resolve_path_for_madx(self.output_dir / self.twiss_out) 

578 madx_script += ( 

579 f'exec, do_twiss_elements(LHCB{accel.beam}, "{twiss_out_path}", {ORBIT_DPP});\n' 

580 ) 

581 return madx_script 

582 

583 def prepare_run(self) -> None: 

584 # As the matched/corrected model is created in the same directory as the original model, 

585 # we do not need to prepare as much. 

586 self.check_accelerator_instance() 

587 LOGGER.debug("Preparing model creation structure") 

588 macros_path = self.accel.model_dir / MACROS_DIR 

589 if not macros_path.exists(): 

590 raise AcceleratorDefinitionError( 

591 f"Folder for the macros does not exist at {macros_path!s}." 

592 ) 

593 

594 @property 

595 def files_to_check(self) -> list[str]: 

596 return [self.twiss_out, self.jobfile, self.logfile]
