39class SpsModelCreator(ModelCreator, ABC):

43 def check_accelerator_instance(self) -> None: 

44 accel: Sps = self.accel 

45 accel.verify_object() # should have been done anyway, but cannot hurt (jdilly) 

46 

47 # Creator specific checks 

48 if accel.acc_model_path is None: 

49 raise AcceleratorDefinitionError( 

50 "The accelerator definition is incomplete: " 

51 "SPS model creation only works with acc-models, but `acc-model-path` was not set. " 

52 ) 

53 

54 # vvv --- same as in LHC, maybe merge? (jdilly, 2025) 

55 if accel.model_dir is None: 

56 raise AcceleratorDefinitionError( 

57 "The accelerator definition is incomplete: " 

58 "Model directory (outputdir option) was not given." 

59 ) 

60 

61 if accel.modifiers is None or not len(accel.modifiers): 

62 raise AcceleratorDefinitionError( 

63 "The accelerator definition is incomplete: No modifiers could be found. " 

64 "Hint: If the accelerator class was instantiated from a model dir, " 

65 f"make sure {JOB_MODEL_MADX_NOMINAL} exists and contains the '{MODIFIER_TAG}' tag." 

66 ) 

67 

68 # hint: if modifiers are given as absolute paths: `path / abs_path` returns `abs_path` (jdilly) 

69 # hint: modifiers were at this point probably already checked (and adapted) in `prepare_run()`. 

70 inexistent_modifiers = [m for m in accel.modifiers if not (accel.model_dir / m).exists()] 

71 if len(inexistent_modifiers): 

72 raise AcceleratorDefinitionError( 

73 "The following modifier files do not exist: " 

74 f"{', '.join([str(accel.model_dir / modifier) for modifier in inexistent_modifiers])}" 

75 ) 

76 

77 # ^^^ --- same as in LHC 

78 

79 # Final check to have at least one strength-file (but only warn user) 

80 if not any(Path(m).suffix == ".str" for m in accel.modifiers): 

81 LOGGER.warning( 

82 "None of the modifiers given ends in '.str', which is kind of expected for the SPS model creation." 

83 )

85 def prepare_options(self, opt) -> bool: 

86 """Use the fetcher to list choices if requested.""" 

87 accel: Sps = self.accel 

88 

89 if opt.fetch == Fetcher.PATH: 

90 if opt.path is None: 

91 raise AcceleratorDefinitionError("Path fetcher chosen, but no path proivided.") 

92 acc_model_path = Path(opt.path) 

93 

94 elif opt.fetch == Fetcher.AFS: 

95 # list 'year' choices --- 

96 acc_model_path = check_folder_choices( 

97 AFS_ACCELERATOR_MODEL_REPOSITORY / Sps.NAME, 

98 msg="No optics tag (flag --year) given", 

99 selection=accel.year, 

100 list_choices=opt.list_choices, 

101 predicate=Path.is_dir, 

102 ) # raises AcceleratorDefintionError if not valid choice 

103 else: 

104 LOGGER.warning( 

105 f"{accel.NAME} model creation is expected to run via a fetcher these days. " 

106 "If you have set the acc_model_path otherwise manually, this might all be correct " 

107 "and you can ignore this warning. Otherwise you will soon run into " 

108 "a MAD-X error. In this case, please provide a fetcher for the model via --fetch flag. " 

109 ) 

110 return 

111 

112 if opt.list_choices: 

113 # all modifiers are actually checked against existing in the 

114 # prepare_run() function. So here we can simply list the strengths files. 

115 check_folder_choices( 

116 acc_model_path / STRENGTHS_SUBDIR, 

117 msg="No/Unknown strength file (flag --str_file) selected", 

118 selection=None, 

119 list_choices=opt.list_choices, 

120 predicate=get_check_suffix_func(".str"), 

121 ) 

122 

123 # Set the found paths --- 

124 accel.acc_model_path = acc_model_path

126 def get_base_madx_script(self) -> str: 

127 accel: Sps = self.accel 

128 use_excitation = accel.excitation != AccExcitationMode.FREE 

129 seq_path = self.resolve_path_for_madx(accel.acc_model_path / "sps.seq") 

130 

131 # The very basics --- 

132 madx_script = ( 

133 "! Load Base Sequence and Strengths/Modifiers ---\n" 

134 "option, -echo; ! suppress output from base sequence loading to keep the log small\n\n" 

135 f"call, file = '{seq_path}';\n" 

136 ) 

137 

138 if accel.modifiers is not None: # includes the strengths file 

139 for modifier in accel.modifiers: 

140 modifier_path = self.resolve_path_for_madx(modifier) 

141 madx_script += f"call, file = '{modifier_path}'; {MODIFIER_TAG}\n" 

142 

143 toolkit_path = self.resolve_path_for_madx(accel.acc_model_path / "toolkit" / "macro.madx") 

144 madx_script += ( 

145 f"call, file ='{toolkit_path}';\n" 

146 "option, echo;\n\n" 

147 "! Create Beam ---\n" 

148 "beam;\n\n" 

149 "! Prepare Tunes ---\n" 

150 f"qx0={accel.nat_tunes[0]:.3f};\n" 

151 f"qy0={accel.nat_tunes[1]:.3f};\n\n" 

152 ) 

153 

154 # Install AC dipole --- 

155 if use_excitation or accel.drv_tunes is not None: 

156 # allow user to modify script and enable excitation, if driven tunes are given 

157 madx_script += ( 

158 f"qxd={accel.drv_tunes[0] % 1:.3f};\n" 

159 f"qyd={accel.drv_tunes[1] % 1:.3f};\n\n" 

160 "! Prepare ACDipole Elements ---\n" 

161 f"use_acd={use_excitation:d}; ! Switch the use of AC dipole\n\n" 

162 "hacmap21 = 0;\n" 

163 "vacmap43 = 0;\n" 

164 "hacmap: matrix, l=0, rm21 := hacmap21;\n" 

165 "vacmap: matrix, l=0, rm43 := vacmap43;\n\n" 

166 "ZKH_MARKER: marker;\n" 

167 "ZKV_MARKER: marker;\n\n" 

168 ) 

169 

170 madx_script += "! Cycle Sequence ---\nseqedit, sequence=sps;\n flatten;\n" 

171 

172 if use_excitation or accel.drv_tunes is not None: 

173 # allow user to modify script and enable excitation, if driven tunes are given 

174 madx_script += ( 

175 "\n" 

176 " ! Install AC dipole ---\n" 

177 " if(use_acd == 1){\n" 

178 " replace, element=ZKHA.21991, by=ZKH_MARKER;\n" 

179 " replace, element=ZKV.21993, by=ZKV_MARKER;\n" 

180 " install, element=hacmap, at=0.0, from=ZKH_MARKER;\n" 

181 " install, element=vacmap, at=0.0, from=ZKV_MARKER;\n" 

182 " }\n\n" 

183 ) 

184 

185 if self._start_bpm is not None: 

186 # I thought cycling to markers has less side-effects, but it seems it doesn't matter. 

187 # If this is anyway useful, we can add it. (jdilly, 2025) 

188 # marker_name = f"OMC_MARKER_{self._start_bpm}" 

189 # madx_script += ( 

190 # f" {marker_name}: marker;\n" 

191 # f" install, element={marker_name}, at=-{self._start_bpm}->L, from={self._start_bpm};\n" 

192 # f" cycle, start = {marker_name};\n" 

193 # ) 

194 madx_script += f" cycle, start = {self._start_bpm};\n" 

195 

196 madx_script += "endedit;\nuse, sequence=sps;\n\n" 

197 

198 # Match tunes --- 

199 madx_script += ( 

200 "! Match Tunes ---\nexec, sps_match_tunes(qx0,qy0);\n\n" 

201 # "twiss, file = 'sps.tfs';\n" # also not sure if needed 

202 ) 

203 return madx_script

205 def get_madx_script(self) -> str: 

206 accel: Sps = self.accel 

207 use_excitation = accel.excitation != AccExcitationMode.FREE 

208 

209 madx_script = self.get_base_madx_script() 

210 twiss_path = self.resolve_path_for_madx(self.output_dir / TWISS_DAT) 

211 twiss_elements_path = self.resolve_path_for_madx(self.output_dir / TWISS_ELEMENTS_DAT) 

212 madx_script += ( 

213 "! Create twiss data files ---\n" 

214 f"{self._get_select_command(pattern=accel.RE_DICT[AccElementTypes.BPMS])}" 

215 f"twiss, file = '{twiss_path}';\n" 

216 "\n" 

217 f"{self._get_select_command()}" 

218 f"twiss, file = '{twiss_elements_path}';\n" 

219 ) 

220 

221 if use_excitation or accel.drv_tunes is not None: 

222 # allow user to modify script and enable excitation, if driven tunes are given 

223 twiss_ac_path = self.resolve_path_for_madx(self.output_dir / TWISS_AC_DAT) 

224 madx_script += ( 

225 f"if(use_acd == 1){{\n" 

226 " betxac = table(twiss, hacmap, betx);\n" 

227 " betyac = table(twiss, vacmap, bety);\n" 

228 f" hacmap21 := 2*(cos(2*pi*qxd)-cos(2*pi*qx0))/(betxac*sin(2*pi*qx0));\n" 

229 f" vacmap43 := 2*(cos(2*pi*qyd)-cos(2*pi*qy0))/(betyac*sin(2*pi*qy0));\n" 

230 "\n" 

231 f"{self._get_select_command(pattern=accel.RE_DICT[AccElementTypes.BPMS], indent=4)}" 

232 f" twiss, file = '{twiss_ac_path}';\n" 

233 f"}}\n" 

234 ) 

235 return madx_script
