42 def check_accelerator_instance(self) -> None: 

43 accel: Sps = self.accel 

44 accel.verify_object() # should have been done anyway, but cannot hurt (jdilly) 

45 

46 # Creator specific checks 

47 if accel.acc_model_path is None: 

48 raise AcceleratorDefinitionError( 

49 "The accelerator definition is incomplete: " 

50 "SPS model creation only works with acc-models, but `acc-model-path` was not set. " 

51 ) 

52 

53 # vvv --- same as in LHC, maybe merge? (jdilly, 2025) 

54 if accel.model_dir is None: 

55 raise AcceleratorDefinitionError( 

56 "The accelerator definition is incomplete: " 

57 "Model directory (outputdir option) was not given." 

58 ) 

59 

60 if accel.modifiers is None or not len(accel.modifiers): 

61 raise AcceleratorDefinitionError( 

62 "The accelerator definition is incomplete: No modifiers could be found. " 

63 "Hint: If the accelerator class was instantiated from a model dir, " 

64 f"make sure {JOB_MODEL_MADX_NOMINAL} exists and contains the '{MODIFIER_TAG}' tag." 

65 ) 

66 

67 # hint: if modifiers are given as absolute paths: `path / abs_path` returns `abs_path` (jdilly) 

68 # hint: modifiers were at this point probably already checked (and adapted) in `prepare_run()`. 

69 inexistent_modifiers = [ 

70 m for m in accel.modifiers if not (accel.model_dir / m).exists()] 

71 if len(inexistent_modifiers): 

72 raise AcceleratorDefinitionError( 

73 "The following modifier files do not exist: " 

74 f"{', '.join([str(accel.model_dir / modifier) for modifier in inexistent_modifiers])}" 

75 ) 

76 

77 # ^^^ --- same as in LHC 

78 

79 # Final check to have at least one strength-file (but only warn user) 

80 if not any(Path(m).suffix == ".str" for m in accel.modifiers): 

81 LOGGER.warning( 

82 "None of the modifiers given ends in '.str', which is kind of expected for the SPS model creation." 

83 )

85 def prepare_options(self, opt) -> bool: 

86 """ Use the fetcher to list choices if requested. """ 

87 accel: Sps = self.accel 

88 

89 if opt.fetch == Fetcher.PATH: 

90 if opt.path is None: 

91 raise AcceleratorDefinitionError( 

92 "Path fetcher chosen, but no path proivided." 

93 ) 

94 acc_model_path = Path(opt.path) 

95 

96 elif opt.fetch == Fetcher.AFS: 

97 # list 'year' choices --- 

98 acc_model_path = check_folder_choices( 

99 AFS_ACCELERATOR_MODEL_REPOSITORY / Sps.NAME, 

100 msg="No optics tag (flag --year) given", 

101 selection=accel.year, 

102 list_choices=opt.list_choices, 

103 predicate=Path.is_dir 

104 ) # raises AcceleratorDefintionError if not valid choice 

105 else: 

106 LOGGER.warning( 

107 f"{accel.NAME} model creation is expected to run via a fetcher these days. " 

108 "If you have set the acc_model_path otherwise manually, this might all be correct " 

109 "and you can ignore this warning. Otherwise you will soon run into " 

110 "a MAD-X error. In this case, please provide a fetcher for the model via --fetch flag. " 

111 ) 

112 return 

113 

114 if opt.list_choices: 

115 # all modifiers are actually checked against existing in the 

116 # prepare_run() function. So here we can simply list the strengths files. 

117 check_folder_choices( 

118 acc_model_path / STRENGTHS_SUBDIR, 

119 msg="No/Unknown strength file (flag --str_file) selected", 

120 selection=None, 

121 list_choices=opt.list_choices, 

122 predicate=get_check_suffix_func(".str") 

123 ) 

124 

125 # Set the found paths --- 

126 accel.acc_model_path = acc_model_path

128 def get_base_madx_script(self): 

129 accel: Sps = self.accel 

130 use_excitation = accel.excitation != AccExcitationMode.FREE 

131 

132 # The very basics --- 

133 madx_script = ( 

134 "! Load Base Sequence and Strengths/Modifiers ---\n" 

135 "option, -echo; ! suppress output from base sequence loading to keep the log small\n\n" 

136 f"call, file = '{accel.acc_model_path / 'sps.seq'!s}';\n" 

137 ) 

138 

139 if accel.modifiers is not None: # includes the strengths file 

140 for modifier in accel.modifiers: 

141 madx_script += f"call, file = '{accel.acc_model_path / modifier!s}'; {MODIFIER_TAG}\n" 

142 

143 madx_script += ( 

144 f"call, file ='{accel.acc_model_path / 'toolkit' / 'macro.madx'!s}';\n" 

145 "option, echo;\n\n" 

146 "! Create Beam ---\n" 

147 "beam;\n\n" 

148 "! Prepare Tunes ---\n" 

149 f"qx0={accel.nat_tunes[0]:.3f};\n" 

150 f"qy0={accel.nat_tunes[1]:.3f};\n\n" 

151 ) 

152 

153 # Install AC dipole --- 

154 if use_excitation or accel.drv_tunes is not None: 

155 # allow user to modify script and enable excitation, if driven tunes are given 

156 madx_script += ( 

157 f"qxd={accel.drv_tunes[0]%1:.3f};\n" 

158 f"qyd={accel.drv_tunes[1]%1:.3f};\n\n" 

159 "! Prepare ACDipole Elements ---\n" 

160 f"use_acd={use_excitation:d}; ! Switch the use of AC dipole\n\n" 

161 "hacmap21 = 0;\n" 

162 "vacmap43 = 0;\n" 

163 "hacmap: matrix, l=0, rm21 := hacmap21;\n" 

164 "vacmap: matrix, l=0, rm43 := vacmap43;\n\n" 

165 "ZKH_MARKER: marker;\n" 

166 "ZKV_MARKER: marker;\n\n" 

167 ) 

168 

169 madx_script += ( 

170 "! Cycle Sequence ---\n" 

171 "seqedit, sequence=sps;\n" 

172 " flatten;\n" 

173 ) 

174 

175 if use_excitation or accel.drv_tunes is not None: 

176 # allow user to modify script and enable excitation, if driven tunes are given 

177 madx_script += ( 

178 "\n" 

179 " ! Install AC dipole ---\n" 

180 " if(use_acd == 1){\n" 

181 " replace, element=ZKHA.21991, by=ZKH_MARKER;\n" 

182 " replace, element=ZKV.21993, by=ZKV_MARKER;\n" 

183 " install, element=hacmap, at=0.0, from=ZKH_MARKER;\n" 

184 " install, element=vacmap, at=0.0, from=ZKV_MARKER;\n" 

185 " }\n\n" 

186 ) 

187 

188 if self._start_bpm is not None: 

189 # I thought cycling to markers has less side-effects, but it seems it doesn't matter. 

190 # If this is anyway useful, we can add it. (jdilly, 2025) 

191 # marker_name = f"OMC_MARKER_{self._start_bpm}" 

192 # madx_script += ( 

193 # f" {marker_name}: marker;\n" 

194 # f" install, element={marker_name}, at=-{self._start_bpm}->L, from={self._start_bpm};\n" 

195 # f" cycle, start = {marker_name};\n" 

196 # ) 

197 madx_script += ( 

198 f" cycle, start = {self._start_bpm};\n" 

199 ) 

200 

201 madx_script += ( 

202 "endedit;\n" 

203 "use, sequence=sps;\n\n" 

204 ) 

205 

206 # Match tunes --- 

207 madx_script += ( 

208 "! Match Tunes ---\n" 

209 "exec, sps_match_tunes(qx0,qy0);\n\n" 

210 # "twiss, file = 'sps.tfs';\n" # also not sure if needed 

211 ) 

212 return madx_script

214 def get_madx_script(self): 

215 accel: Sps = self.accel 

216 use_excitation = accel.excitation != AccExcitationMode.FREE 

217 

218 madx_script = self.get_base_madx_script() 

219 madx_script += ( 

220 "! Create twiss data files ---\n" 

221 f"{self._get_select_command(pattern=accel.RE_DICT[AccElementTypes.BPMS])}" 

222 f"twiss, file = {accel.model_dir / TWISS_DAT};\n" 

223 "\n" 

224 f"{self._get_select_command()}" 

225 f"twiss, file = {accel.model_dir / TWISS_ELEMENTS_DAT};\n" 

226 ) 

227 

228 if use_excitation or accel.drv_tunes is not None: 

229 # allow user to modify script and enable excitation, if driven tunes are given 

230 madx_script += ( 

231 f"if(use_acd == 1){{\n" 

232 " betxac = table(twiss, hacmap, betx);\n" 

233 " betyac = table(twiss, vacmap, bety);\n" 

234 f" hacmap21 := 2*(cos(2*pi*qxd)-cos(2*pi*qx0))/(betxac*sin(2*pi*qx0));\n" 

235 f" vacmap43 := 2*(cos(2*pi*qyd)-cos(2*pi*qy0))/(betyac*sin(2*pi*qy0));\n" 

236 "\n" 

237 f"{self._get_select_command(pattern=accel.RE_DICT[AccElementTypes.BPMS], indent=4)}" 

238 f" twiss, file = {accel.model_dir / TWISS_AC_DAT};\n" 

239 f"}}\n" 

240 ) 

241 return madx_script
