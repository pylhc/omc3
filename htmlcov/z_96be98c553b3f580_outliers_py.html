22def get_filter_mask(data: ArrayLike, x_data: ArrayLike = None, limit: float = 0.0, 

23 niter: int = 20, nsig: int = None, mask: ArrayLike = None) -> ArrayLike: 

24 r""" 

25 Filters the array of values which are meant to be constant or a linear function of the x-data 

26 array if that is provided, by checking how many sigmas they are deviating from the average. 

27 

28 The outlier filtering function is utilized at multiple stages of the data analysis. 

29 It removes data points in the tails of the measured distribution, 

30 which are too populated due to the finite sample size, 

31 assuming a normal distribution specified by measured mean and standard 

32 deviation of the given data. 

33 A data point, outside of a user-specified limit, 

34 is removed if there is less than a 50% chance 

35 that it is from the specified normal distribution. 

36 

37 In particular: 

38 The function gets an array :math:`y` of data of length :math:`n_y`, 

39 which can be any scalar data and is currently used for e.g. tune data 

40 measured per BPM or a time series from the BBQ. 

41 In addition, a cleaning ``limit`` can be given, 

42 inside which the data points are always kept. 

43 Further an optional array :math:`x` (via the parameter ``x_data``) 

44 can be given, in which case a linear fit :math:`y = ax + c` is attempted 

45 to remove any slope on the data, before calculating momenta on the data. 

46 An upper boundary in :math:`n_\sigma` is then determined from the 

47 percent point function (ppf) of a Student's t-distribution, 

48 given :math:`n_x` degrees of freedom at :math:`100\% - 50\% / n_y`. 

49 Iteratively, mean :math:`\left< y \right>` and standard deviation :math:`\sigma` 

50 of the data is calculated and only data within :math:`n_\sigma \cdot \sigma`, 

51 or the given cleaning limit, around :math:`\left< y \right>` 

52 is kept for the next iteration. The loop stops either after 20 iterations, 

53 when there is no more data outside the boundaries or 

54 when there are less than three data points remaining. 

55 

56 Returns a filter mask for the original array (meaning ``True`` for elements that should be kept). 

57 

58 Args: 

59 data (ArrayLike): Data to filter. 

60 x_data (ArrayLike): Additional x-data. If given a linear function is fit on the data, 

61 and the outliers filtered on the difference to this function. 

62 Otherwise, the distribution of data is used directly for filtering. 

63 limit (float): Only data deviating more than this limit from the average is filtered. 

64 niter (int): Maximum number of filter iterations to do. 

65 Iterations are interrupted if the number of datapoints did not shrink 

66 in the last iteration or if there is only two or less data-points left. 

67 nsig (int): number of sigmas within the data points are considered okay, outside they are 

68 considered outliers. If not given, nsigmas is got from the 

69 1-0.5/length(data) percentile of a student-t distribution. 

70 mask (ArrayLike): Boolean mask of data-points to consider. 

71 If ``None``, all data is considered. 

72 

73 Returns: 

74 ArrayLike: Boolean array containing ``True`` entries for data-points that are good, 

75 and ``False`` for the ones that should be filtered. 

76 

77 """ 

78 LOGGER.debug("Creating Outlier-Filter mask.") 

79 

80 if x_data is not None and len(data) != len(x_data): 

81 raise ValueError("Datasets are not equally long.") 

82 

83 if mask is not None: 

84 if not len(data) == len(mask): 

85 raise ValueError("Mask is not equally long as dataset.") 

86 else: 

87 mask = np.ones_like(data, dtype=bool) 

88 

89 if nsig is None: 

90 nsig = _get_significance_cut_from_length(np.sum(mask)) 

91 

92 # Set number of remaining points to check decrease in loop 

93 n_previous = np.sum(mask) + 1 # initialization for the first check 

94 

95 # Cleaning iteration 

96 for _ in range(niter): 

97 # check that number of points decreases and some are left 

98 n_current = np.sum(mask) 

99 if not ((n_current < n_previous) and (n_current > 2)): 

100 break 

101 n_previous = n_current 

102 

103 # get the (remaining) data-points used to get moments from, 

104 # if x-data is given, fit and remove slope. 

105 if x_data is None: 

106 y, y_orig = data[mask], data 

107 else: 

108 y, y_orig = _get_data_without_slope(mask, x_data, data) 

109 

110 # keep only points within nsig sigma of the remaining data. 

111 avg, std = np.mean(y), np.std(y) 

112 mask = np.logical_and(mask, np.abs(y_orig - avg) < np.max([limit, nsig * std])) 

113 else: 

114 LOGGER.debug("Outlier Filter loop exceeds maximum number of iterations." 

115 " Current filter-mask will be used.") 

116 return mask

119def _get_data_without_slope(mask, x, y): 

120 """ Remove the slope on the data by performing a linear fit. """ 

121 import warnings 

122 try: 

123 from numpy.exceptions import RankWarning as IgnoredWarning 

124 except ImportError: # it does not exist on numpy versions supporting 3.9 

125 IgnoredWarning = Warning # noqa: N806 (it is a class) 

126 

127 # We filter out the "Polyfit may be poorly conditioned" warning that can happen 

128 # during the polyfit. We relay it as a logged message, which allows us to avoid 

129 # polluting the stderr and allows the user to not see it depending on log level 

130 with warnings.catch_warnings(record=True) as records: 

131 warnings.simplefilter("ignore", category=IgnoredWarning) 

132 m, b = np.polyfit(x[mask], y[mask], 1) 

133 

134 # We log any captured warning at warning level 

135 for warning in records: 

136 LOGGER.warning(f"Polyfit warning: {warning.message}") 

137 

138 return y[mask] - b - m * x[mask], y - b - m * x

141def _get_significance_cut_from_length(length): 

142 """ Set the sigma cut, that expects one value to be cut 

143 if it is sample of normal distribution.""" 

144 return t.ppf(1 - 0.5 / length, length)
