1""" 

2Response MAD-NG 

3=============== 

4 

5Provides a function to create the responses of beta, phase, dispersion, tune and coupling via MAD-NG derivatives. 

6 

7The variables under investigation need to be provided as a list (which can be obtained from the accelerator 

8class). 

9 

10For now, the response matrix is stored in a hdf5 file. 

11 

12Differences to the MAD-X implementation 

13--------------------------------------- 

14 

15Differential Algebra (DA) is used to compute the derivatives of the optics parameters with respect to the 

16variables (knobs). For most optics parameters (beta, phase, dispersion), a single twiss calculation is performed 

17with DA enabled, allowing the extraction of derivatives directly from the twiss output. 

18 

19Coupling derivatives are computed in a separate twiss calculation after perturbing the variables slightly. 

20This is necessary because the coupling calculation is unstable when there is no coupling present. 

21By applying a small perturbation to the variables, we ensure that the coupling derivatives can be computed reliably. 

22This perturbation is a constant value added to all variables during the coupling twiss calculation, and has been tested 

23for the LHC to provide the most stable results. 

24 

25By only performing two twiss calculations (one for standard optics and one for coupling), 

26this approach is significantly more efficient than performing separate calculations for each variable. 

27 

28:author: Joshua Gray 

29"""

88@contextmanager 

89def _initialise_madng_with_logging(accel_inst: Accelerator): 

90 """Initialise MAD-NG, sending the stdout of the MAD-NG instance to a log file. 

91 

92 This function does the very basics; 

93 - Creates a MAD-NG instance with logging to a file 

94 - Loads the sequence 

95 - Adds the beam to the sequence 

96 This is a minimal setup that allows the MAD-NG instance to be used for this sequence. 

97 

98 Args: 

99 accel_inst: Accelerator instance 

100 

101 Returns: 

102 Tuple of (MAD instance, log file path) 

103 """ 

104 # Create log file in the model directory 

105 log_path = accel_inst.model_dir / "madng_response.log" 

106 LOGGER.info(f"MAD-NG output will be logged to: {log_path}") 

107 

108 # Initialize MAD-NG with stdout redirected to log file 

109 with MAD(stdout=str(log_path), redirect_stderr=True, debug=True) as mad: 

110 _load_sequence(mad, accel_inst) 

111 _setup_beam(mad, accel_inst.energy) 

112 

113 yield mad, log_path

130def _load_sequence(mad: MAD, accel_inst: Accelerator): 

131 """Load a sequence into MAD-NG, handling all path creation, file generation, and cleanup internally.""" 

132 # Create the model creator and prepare 

133 creator_class = get_model_creator_class(accel_inst, CreatorType.NOMINAL) 

134 creator = creator_class(accel_inst) 

135 creator.prepare_run() 

136 

137 # Construct paths 

138 madx_seq_path = accel_inst.model_dir / creator.save_sequence_filename 

139 seq_name = creator.sequence_name 

140 log_file = accel_inst.model_dir / "madng_sequence_creation.log" 

141 

142 if madx_seq_path.exists(): 

143 raise FileExistsError(f"Saved sequence file {madx_seq_path} already exists!") 

144 

145 # Generate the MAD-X sequence file 

146 madx_string = creator.get_base_madx_script() + "\n" + creator.get_save_sequence_script() 

147 run_string(madx_string, log_file=log_file, cwd=accel_inst.model_dir) 

148 

149 if not madx_seq_path.exists(): 

150 raise FileNotFoundError( 

151 f"Saved sequence file {madx_seq_path} was not created! Check {log_file}" 

152 ) 

153 

154 # Load into MAD-NG 

155 mad.send(f'MADX:load("{str(madx_seq_path.absolute())}")') 

156 if mad.MADX[seq_name] == 0: 

157 raise ValueError( 

158 f"Sequence '{seq_name}' not found in MAD file '{str(madx_seq_path.absolute())}'" 

159 ) 

160 mad.send(f"loaded_sequence = MADX.{seq_name}") 

161 mad.loaded_sequence.dir = accel_inst.beam_direction 

162 

163 # Clean up the temporary file 

164 madx_seq_path.unlink(missing_ok=True)

174def create_fullresponse( 

175 accel_inst: Accelerator, 

176 variable_categories: Sequence[str], 

177) -> dict[str, pd.DataFrame]: 

178 """Generate a dictionary containing response matrices for 

179 beta, phase, dispersion, tune and coupling using MAD-NG derivatives. 

180 

181 Args: 

182 accel_inst : Accelerator Instance. 

183 variable_categories (list): Categories of the variables/knobs to use. 

184 

185 Returns: 

186 dict: Dictionary of response DataFrames keyed by optics type (e.g., 'BETAX', 'PHASEX', etc.) 

187 """ 

188 LOGGER.info("Creating fullresponse via MAD-NG") 

189 

190 # Initialise MAD-NG with logging 

191 with _initialise_madng_with_logging(accel_inst) as (mad, log_path): 

192 # Get variables 

193 variables = accel_inst.get_variables(classes=variable_categories) 

194 if len(variables) == 0: 

195 raise ValueError("No variables found! Make sure your categories are valid!") 

196 

197 LOGGER.info(f"Computing response for {len(variables)} variables") 

198 

199 try: 

200 # Compute derivatives with twiss 

201 response_dict = _compute_response_with_derivatives(mad, variables, accel_inst) 

202 except RuntimeError as e: 

203 LOGGER.error("Error while computing response with MAD-NG derivatives.") 

204 _log_madng_output(log_path) 

205 raise e 

206 _log_madng_output(log_path) 

207 

208 return response_dict

211def _compute_response_with_derivatives( 

212 mad: MAD, variables: list[str], accel_inst: Accelerator 

213) -> dict[str, pd.DataFrame]: 

214 """ 

215 Compute response matrices using MAD-NG derivatives. 

216 

217 This function sets up differential algebra in MAD-NG to compute derivatives of optics 

218 (beta, phase, dispersion, tunes, coupling) with respect to the given variables (knobs). 

219 It performs two twiss calculations: one for standard optics and one for coupling, 

220 then extracts and processes the derivative data into response matrices. 

221 

222 Args: 

223 mad: MAD-NG instance 

224 variables: List of variable names (knobs) for which to compute responses 

225 accel_inst: Accelerator instance providing BPM patterns and other info 

226 

227 Returns: 

228 Dictionary of response DataFrames keyed by optics type (e.g., 'BETAX', 'PHASEX', etc.) 

229 """ 

230 n_vars = len(variables) 

231 

232 # Define optics parameters for which we want derivatives 

233 optics_list = list(MADNG_VARMAP.keys()) 

234 coupling_params = list(COUPLING_VARMAP.keys()) 

235 

236 # Create mapping from optics parameters to derivative column names, e.g., 'betx' -> ['betx_100', 'betx_010', ...] 

237 kopt_dict = _create_kopt_dict(optics_list, n_vars) 

238 flat_opt_list = [item for sublist in kopt_dict.values() for item in sublist] 

239 

240 # Create mapping for coupling parameters e.g., 'f1001' -> ['f1001_100', 'f1001_010', ...] 

241 coupling_kopt_dict = _create_kopt_dict(coupling_params, n_vars) 

242 coupling_optics_list = [item for sublist in coupling_kopt_dict.values() for item in sublist] 

243 

244 dk = dict.fromkeys(variables, NG_DELTA_K) 

245 

246 # Set up Differential Algebra for derivatives 

247 # This MAD-NG script performs the following workflow: 

248 # 1. Receive lists of knobs, optics parameters, and BPM pattern from Python. 

249 # 2. Create a differential algebra map (damap) for containing the variables and initial conditions. 

250 # 3. Convert knobs to DA variables by adding their damap components. 

251 # 4. Select BPMs and END element for observation. 

252 # 5. Run twiss calculation with derivatives enabled for standard optics. 

253 # 6. Receive delta k values and coupling optics list. 

254 # 7. Perturb knobs with delta k for stable coupling derivative computation. 

255 # 8. Run second twiss calculation for coupling derivatives. 

256 mad.send(""" 

257--start-mad 

258local knob_list = py:recv() 

259local opt_list = py:recv() 

260local bpm_pattern = py:recv() 

261local num_k = #knob_list 

262local k_ord = 2 

263local x0 = MAD.damap { nv = 6, np = num_k, no = {k_ord, k_ord, k_ord, k_ord, 1, 1}, po=1, pn=knob_list} 

264for i, knob in ipairs(knob_list) do 

265 ! dk[knob] = MADX[knob] + 0 (Converts the knob to a DA variable) 

266 MADX[knob] = MADX[knob] + x0[knob] 

267end 

268 

269-- Observe all elements 

270local observed in MAD.element.flags 

271local drift_element in MAD.element 

272loaded_sequence:deselect(observed) 

273loaded_sequence:select(observed, {pattern=bpm_pattern}) ! Observe all BPMs 

274loaded_sequence:select(observed, {pattern="$end"}) ! Also observe END marker for tunes 

275 

276-- Compute twiss with derivatives 

277tws, _ = twiss { 

278 sequence=loaded_sequence, 

279 observe=1, 

280 X0=x0, 

281 trkopt=opt_list, 

282 coupling=true, ! Set to true for betx and bety calculation 

283} 

284 

285local dk = py:recv() 

286local coupling_list = py:recv() 

287-- Apply delta k to knobs so that the coupling derivatives can be computed too 

288for i, knob in ipairs(knob_list) do 

289 MADX[knob] = MADX[knob] + dk[knob] 

290end 

291 

292tws_coupling, _ = twiss { 

293 sequence=loaded_sequence, 

294 observe=1, 

295 X0=x0, 

296 trkopt=coupling_list, 

297 coupling=true, 

298} 

299--end-mad 

300""") 

301 

302 # Send the required data to MAD-NG script 

303 mad.send(variables) # List of knob names 

304 mad.send(flat_opt_list) # Flattened list of derivative column names for standard optics 

305 mad.send(accel_inst.RE_DICT[AccElementTypes.BPMS]) # BPM pattern for observation 

306 mad.send(dk) # Dictionary of delta k values for perturbation 

307 mad.send(coupling_optics_list) # Flattened list of derivative column names for coupling 

308 

309 # Extract and process twiss data from MAD-NG 

310 optics_columns = [MADNG_OPTICS.get(optic, optic) for optic in optics_list] 

311 all_cols = ["name"] + optics_columns + flat_opt_list 

312 

313 # Get twiss data with all columns 

314 LOGGER.debug(f"Extracting {all_cols} from MAD-NG twiss output") 

315 twiss_df = mad.tws.to_df(columns=all_cols) 

316 twiss_df = twiss_df.set_index("name") 

317 LOGGER.debug(f"Twiss extracted with shape {twiss_df.shape}") 

318 

319 LOGGER.debug(f"Extracting {coupling_optics_list} from MAD-NG twiss output") 

320 twiss_df_coupling = mad.tws_coupling.to_df(columns=["name"] + coupling_optics_list) 

321 twiss_df_coupling = twiss_df_coupling.set_index("name") 

322 LOGGER.debug(f"Coupling twiss extracted with shape {twiss_df_coupling.shape}") 

323 

324 # Extract response matrices 

325 response_dict = {} 

326 

327 # Get element names 

328 bpms = twiss_df.index 

329 

330 # Tunes (at the END element) 

331 end = bpms[-1] # Get the 'END' element for tunes 

332 tune_data = np.array( 

333 [ 

334 twiss_df[kopt_dict["mu1"]].loc[end].to_numpy(), 

335 twiss_df[kopt_dict["mu2"]].loc[end].to_numpy(), 

336 ] 

337 ) 

338 response_dict[TUNE] = pd.DataFrame( 

339 data=tune_data, index=[f"{TUNE}1", f"{TUNE}2"], columns=variables 

340 ) 

341 

342 # Remove END from bpms for other optics 

343 bpms = bpms[:-1] 

344 twiss_df = twiss_df.loc[bpms] 

345 twiss_df_coupling = twiss_df_coupling.loc[bpms] 

346 

347 # Build response dictionary for other optics 

348 for optic_param, response_key in MADNG_VARMAP.items(): 

349 deriv_cols = kopt_dict[optic_param] 

350 response_dict[response_key] = pd.DataFrame( 

351 data=twiss_df[deriv_cols].to_numpy(), # shape (bpms, n_vars) 

352 index=bpms, 

353 columns=variables, 

354 ) 

355 

356 # Compute normalised dispersion response 

357 # d/dx (D/sqrt(beta)) = (2 beta * dD/dx - D * dbeta/dx) / (2 beta^(3/2)) 

358 # Normalised Y dispersion can't even be used, see global correction OPTICS_PARAMS_CHOICES, so we skip it 

359 for plane in ["X"]: 

360 disp_col = f"{DISPERSION}{plane}" 

361 beta_col = f"{BETA}{plane}" 

362 # Convert to numpy arrays for computation and avoid NaNs (pandas may introduce NaNs) 

363 disp = twiss_df[disp_col.lower()].to_numpy() 

364 beta = twiss_df[beta_col.lower()].to_numpy() 

365 disp_resp = response_dict[disp_col].to_numpy() 

366 beta_resp = response_dict[beta_col].to_numpy() 

367 

368 numerator = 2 * beta[:, np.newaxis] * disp_resp - disp[:, np.newaxis] * beta_resp 

369 denominator = 2 * beta[:, np.newaxis] ** 1.5 

370 norm_disp_resp = numerator / denominator 

371 

372 response_dict[f"{NORM_DISPERSION}{plane}"] = pd.DataFrame( 

373 data=norm_disp_resp, index=bpms, columns=variables 

374 ) 

375 

376 # Normalise beta response 

377 for plane in ["X", "Y"]: 

378 beta_col = f"{BETA}{plane}" 

379 beta = twiss_df[beta_col.lower()].to_numpy() 

380 response_dict[beta_col] = response_dict[beta_col] / beta[:, np.newaxis] 

381 

382 for coupling_param, response_key in COUPLING_VARMAP.items(): 

383 deriv_cols = coupling_kopt_dict[coupling_param] 

384 coupling_data = twiss_df_coupling[deriv_cols] 

385 response_dict[f"{response_key}R"] = pd.DataFrame( 

386 data=coupling_data.map(lambda x: x.real).to_numpy(), 

387 index=bpms, 

388 columns=variables, 

389 ) 

390 response_dict[f"{response_key}I"] = pd.DataFrame( 

391 data=coupling_data.map(lambda x: x.imag).to_numpy(), 

392 index=bpms, 

393 columns=variables, 

394 ) 

395 

396 if accel_inst.beam_direction == -1: 

397 # Change the sign of the coupling responses for reverse beam direction 

398 # In all honesty, I don't know why this has to be done, but it works... (jgray 2026) 

399 for coupling_param in COUPLING_VARMAP.values(): 

400 response_dict[f"{coupling_param}R"] *= -1 

401 response_dict[f"{coupling_param}I"] *= -1 

402 

403 # Check for any NaNs in the response matrices 

404 for key, df in response_dict.items(): 

405 if df.isna().any().any(): 

406 LOGGER.warning(f"NaNs found in response matrix for {key}: {df.isna().sum().sum()} NaNs") 

407 

408 return response_dict

411def _create_kopt_dict(params: list[str], n_vars: int) -> dict[str, list[str]]: 

412 """ 

413 Create a dictionary mapping each optics parameter to its list of derivative column names. 

414 

415 For each parameter (e.g., 'betx'), this generates a list of column names corresponding to 

416 the derivatives of that parameter with respect to each variable (knob). Each column name 

417 uses a binary mask string to indicate which variable the derivative is taken with respect to. 

418 

419 The binary mask is a string of length n_vars, consisting of '0's and '1's, where a '1' at 

420 position j indicates that the derivative is with respect to the j-th variable in the list 

421 provided to MAD-NG. All other positions are '0'. 

422 

423 For example, with 3 variables ['var1', 'var2', 'var3'] and parameter 'betx', the output would be: 

424 { 

425 'betx': ['betx_100', 'betx_010', 'betx_001'] 

426 } 

427 Here, 'betx_100' is the derivative of betx w.r.t. var1, 'betx_010' w.r.t. var2, etc. 

428 

429 Args: 

430 params: List of parameter names (e.g., ['mu1', 'betx', ...]) 

431 n_vars: Number of variables (knobs) for which derivatives are computed 

432 

433 Returns: 

434 Dictionary mapping each parameter name to its list of derivative column names 

435 """ 

436 return { 

437 param: [f"{param}_{_make_binary_mask(j, n_vars)}" for j in range(n_vars)] 

438 for param in params 

439 }

441 

442def _make_binary_mask(j: int, n_vars: int) -> str: 

443 """ 

444 Create binary mask string for derivative w.r.t. variable j. 

445 

446 Generates a string of '1's and '0's where the j-th position is '1' and others '0', 

447 used by MAD-NG to specify which variable's derivative is being computed. 

448 For example, with 3 variables and j=1, returns '010'. 

449 

450 Args: 

451 j: Index of the variable for which the derivative is taken 

452 n_vars: Total number of variables 

453 

454 Returns: 

455 Binary mask string 

456 """ 

457 return "".join("1" if i == j else "0" for i in range(n_vars))
