28def _get_params(): 

29 params = EntryPointParameters() 

30 params.add_parameter( 

31 name="outputdir", 

32 type=Path, 

33 help="Output path for model, twiss files will be writen here. [Required]", 

34 ) 

35 params.add_parameter( 

36 name="type", 

37 choices=(CreatorType.NOMINAL.value, CreatorType.BEST_KNOWLEDGE.value), # this script manages only these two 

38 default=CreatorType.NOMINAL.value, 

39 help="Type of model to create.", 

40 ) 

41 params.add_parameter( 

42 name="logfile", 

43 type=Path, 

44 help=("Path to the file where to write the MAD-X script output." 

45 "If not provided it will be written to sys.stdout.") 

46 ) 

47 params.add_parameter( 

48 name="show_help", 

49 action="store_true", 

50 help="Instructs the subsequent modules to print a help message" 

51 ) 

52 params.update(get_fetcher_params()) 

53 return params

81@entrypoint(_get_params()) 

82def create_instance_and_model(opt, accel_opt) -> Accelerator | None: 

83 """ 

84 Manager Keyword Args: 

85 *--Required--* 

86 

87 - **accel**: 

88 

89 Choose the accelerator to use.Can be the class already. 

90 

91 choices: ``['lhc', 'ps', 'esrf', 'psbooster', 'skekb', 'JPARC', 'petra', 'iota']`` 

92 

93 

94 Creator Keyword Args: 

95 *--Required--* 

96 

97 - **outputdir** *(str)*: 

98 

99 Output path for model, twiss files will be writen here. 

100 

101 

102 *--Optional--* 

103 

104 - **logfile** *(str)*: 

105 

106 Path to the file where to write the MAD-X script output.If not 

107 provided it will be written to sys.stdout. 

108 

109 

110 - **type**: 

111 

112 Type of model to create. 

113 

114 choices: ``('nominal', 'best_knowledge')`` 

115 

116 

117 Accelerator Keyword Args: 

118 lhc: :mod:`omc3.model.accelerators.lhc` 

119 

120 ps: :mod:`omc3.model.accelerators.ps` 

121 

122 esrf: :mod:`omc3.model.accelerators.esrf` 

123 

124 psbooster: :mod:`omc3.model.accelerators.psbooster` 

125 

126 skekb: :mod:`omc3.model.accelerators.skekb` 

127 

128 iota: :mod:`omc3.model.accelerators.iota` 

129 

130 petra: :mod:`omc3.model.accelerators.petra` (not implemented) 

131 

132 JPARC: Not implemented 

133 """ 

134 # first, if help is requested, gather all help info and print it 

135 if opt.show_help: 

136 try: 

137 #with silence(): 

138 accel_class = model_manager.get_accelerator_class(accel_opt) 

139 print(f"---- Accelerator {accel_class.__name__} | Usage ----\n") 

140 print_help(accel_class.get_parameters()) 

141 except Exception as e: # noqa: BLE001 (ugly that we catch all exceptions here...) 

142 LOGGER.debug(f"An error occurred: {e}") 

143 pass 

144 

145 print("---- Model Creator | Usage ----\n") 

146 print_help(model_manager._get_params()) 

147 print_help(_get_params()) 

148 return None 

149 

150 # proceed to the creator 

151 accel_inst: Accelerator = model_manager.get_accelerator(accel_opt) 

152 require_param("type", _get_params(), opt) 

153 LOGGER.debug(f"Accelerator Instance {accel_inst.NAME}, model type {opt.type}") 

154 

155 # model_dir is used as the output directory 

156 require_param("outputdir", _get_params(), opt) 

157 outputdir = Path(opt.outputdir) 

158 accel_inst.model_dir = outputdir.absolute() 

159 

160 creator_class = get_model_creator_class(accel_inst, opt.type) 

161 creator: ModelCreator = creator_class(accel_inst, logfile=opt.logfile) 

162 

163 # Check if the options (i.e. the values of the arguments given) are valid choices. 

164 # This needs to be done by the creator itself, as it should know what valid options are 

165 # and can then also print them. If this fails, we have to abort. 

166 # 

167 # !! NOTE: If succesfull, THIS CAN MODIFY THE ACCELERATOR INSTANCE on creator.accel !! 

168 try: 

169 creator.prepare_options(opt) 

170 except AcceleratorDefinitionError: 

171 if not opt.list_choices: 

172 raise 

173 return None 

174 

175 # Save config only now, to not being written out for each time the choices are listed 

176 save_config(outputdir, opt=opt, unknown_opt=accel_opt, script=__file__) 

177 

178 # Run the actual model creation 

179 creator.full_run() 

180 

181 return accel_inst 

182

184if __name__ == "__main__": 

185 create_instance_and_model()
