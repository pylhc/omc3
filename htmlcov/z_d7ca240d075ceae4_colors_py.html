44def change_color_brightness(color, amount=0.5): 

45 """ 

46 Lightens the given color by multiplying (1-luminosity) by the given amount. Input can be 

47 ``matplotlib`` color string, hex string, or RGB tuple. An amount of 1 equals to no change. 0 

48 is very bright (white) and 2 is very dark. Original code by Ian Hincks 

49 Source: https://stackoverflow.com/questions/37765197/darken-or-lighten-a-color-in-matplotlib 

50 """ 

51 if not (0<=amount<=2): 

52 raise ValueError("The brightness change has to be between 0 and 2." 

53 f" Instead it was {amount}") 

54 try: 

55 c = mpl.colors.cnames[color] 

56 except KeyError: 

57 c = color 

58 

59 c = colorsys.rgb_to_hls(*mpl.colors.to_rgb(c)) 

60 return colorsys.hls_to_rgb(c[0], max(0, min(1, amount * c[1])), c[2])
